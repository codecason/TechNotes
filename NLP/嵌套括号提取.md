提取包含嵌套{}的最外层{}

~~~python
import re

def brace_pattern(max_depth: int) -> re.Pattern:
    """
    生成支持到 max_depth 层嵌套的花括号匹配正则（仅用 re 标准库）。
    depth=1:  \{[^{}]*\}
    depth=2:  \{(?:[^{}]*|\{[^{}]*\})*\}
    depth=3:  \{(?:[^{}]*|\{(?:[^{}]*|\{[^{}]*\})*\})*\}
    以此类推。
    """
    if max_depth < 1:
        raise ValueError("max_depth 必须 >= 1")

    # depth=1 的模式
    pat = r'\{[^{}]*\}'

    # 递推构造更高层： depth=d 的模式把 depth=d-1 作为内层“原子”
    for _ in range(2, max_depth + 1):
        pat = r'\{(?:[^{}]*|' + pat + r')*\}'

    return re.compile(pat)

def find_blocks(s: str, max_depth: int):
    """
    用给定最大深度的正则，返回匹配到的非重叠花括号块（字符串形式）。
    注意：max_depth 不足时，会退化为匹配更浅层的块，而非真正的“最外层”。
    """
    pattern = brace_pattern(max_depth)
    return [m.group(0) for m in pattern.finditer(s)]

def run_tests():
    # 测试用例：name, input_string, 期望结果（按不同 max_depth）
    cases = [
        (
            "depth2 single",
            "x{a{b}c}y",
            {
                # depth=1 无法吞掉外层，会只命中内层的 {b}
                1: ["{b}"],
                2: ["{a{b}c}"],
                3: ["{a{b}c}"],
            },
        ),
        (
            "depth2 siblings",
            "x{a{b}c}{d{e}f}y",
            {
                1: ["{b}", "{e}"],
                2: ["{a{b}c}", "{d{e}f}"],
                3: ["{a{b}c}", "{d{e}f}"],
            },
        ),
        (
            "depth3 single",
            "x{a{b{c}d}e}y",
            {
                # depth=2 只能匹配到内层的 {b{c}d}（而不是最外层）
                1: ["{c}"],
                2: ["{b{c}d}"],
                3: ["{a{b{c}d}e}"],
            },
        ),
        (
            "depth3 siblings",
            "x{a{b{c}d}e}{f{g{h}i}j}y",
            {
                1: ["{c}", "{h}"],
                2: ["{b{c}d}", "{g{h}i}"],
                3: ["{a{b{c}d}e}", "{f{g{h}i}j}"],
            },
        ),
        (
            "adjacent deep then shallow",
            "{a{b{c}d}e}{x}",
            {
                1: ["{c}", "{x}"],
                2: ["{b{c}d}", "{x}"],
                3: ["{a{b{c}d}e}", "{x}"],
            },
        ),
        (
            "text with newlines",
            "pre\n{a\n{b}\n}post",
            {
                1: ["{b}"],
                2: ["{a\n{b}\n}"],
                3: ["{a\n{b}\n}"],
            },
        ),
    ]

    for depth in (1, 2, 3):
        print(f"\n=== Testing with max_depth={depth} ===")
        for name, s, expected in cases:
            got = find_blocks(s, depth)
            exp = expected[depth]
            print(f"{name:24s} -> {got}")
            assert got == exp, f"失败: {name}, depth={depth}, 期望 {exp}, 实际 {got}"

    print("\n全部断言通过。")

if __name__ == "__main__":
    run_tests()
~~~
